# 2667번 - 단지번호붙이기

**난이도: Silver 1**

## 문제
<그림 1>과 같이 정사각형 모양의 지도가 있다. 1은 집이 있는 곳을, 0은 집이 없는 곳을 나타낸다. 철수는 이 지도를 가지고 연결된 집의 모임인 단지를 정의하고, 단지에 번호를 붙이려 한다. 여기서 연결되었다는 것은 어떤 집이 좌우, 혹은 아래위로 다른 집이 있는 경우를 말한다. 대각선상에 집이 있는 경우는 연결된 것이 아니다. <그림 2>는 <그림 1>을 단지별로 번호를 붙인 것이다. 지도를 입력하여 단지수를 출력하고, 각 단지에 속하는 집의 수를 오름차순으로 정렬하여 출력하는 프로그램을 작성하시오.

## 입력
첫 번째 줄에는 지도의 크기 N(정사각형이므로 가로와 세로의 크기는 같으며 5≤N≤25)이 입력되고, 그 다음 N줄에는 각각 N개의 자료(0혹은 1)가 입력된다.

## 출력
첫 번째 줄에는 총 단지수를 출력하시오. 그리고 각 단지내 집의 수를 오름차순으로 정렬하여 한 줄에 하나씩 출력하시오.

## 예제 입력 1
```
7
0110100
0110101
1110101
0000111
0100000
0111110
0111000
```

## 예제 출력 1
```
3
7
8
9
```

## 소스코드 (C#)
```cs
using System.Diagnostics;
using System.Text;
namespace CodeTesting
{
    internal class Program
    {
        public static int[,] nums;
        public static int[,] visit;
        public static int cnt = 0;
        public static int visitCnt = 0;
        //첫번째가 cnt -> visit에 들어갈 숫자이고, 총 개수
        public static PriorityQueue<int, int> que = new PriorityQueue<int, int>();
        static void Main(string[] args)
        {

            StreamReader reader = new StreamReader(Console.OpenStandardInput());
            StreamWriter writer = new StreamWriter(Console.OpenStandardOutput());
            StringBuilder op = new StringBuilder();
            //nums는 0과1로 이루어진 이차원 배열
            //visit는 각 단지를 적은 visit가 들어있음
            //cnt가 올라갈 때는 for문 시작에 현재 visit가 0이면 cnt가 ++가 되고 거기로 부터 파생되는 칸들은 그 cnt로 변경된다.
            //for문을 다시 전체로 돌려서 dfs해서 확인한다.
            int n = int.Parse(reader.ReadLine());
            visit = new int[n, n];
            nums = new int[n, n];

            for(int i=0; i<n; i++)
            {
                string s = reader.ReadLine();
                for(int j=0; j<n; j++)
                {
                    nums[i, j] = int.Parse(s[j].ToString());
                }
            }

            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    visitCnt = 0;
                    //해당 visit가 방문을 했다면 continue로 넘어가고 방문 이력이 없다면 cnt++를 하고 해당 부분부터 dfs를 실행한다.
                    if (visit[i, j] == 0 && nums[i,j] ==1)
                    {
                        cnt++;
                        DFS(i, j, n);
                       // Console.WriteLine($" {i} {j}  .{cnt} , {visitCnt}");
                        que.Enqueue(cnt, visitCnt);
                    }
                    else { continue; }
                }
            }
            writer.WriteLine(que.Count);
            writer.Flush();
            while (que.Count > 0)
            {
                que.TryDequeue(out int a, out int priority);
                writer.WriteLine(priority);
                writer.Flush();

            }
           

        }
        //기존 for문으로 확인하는 것이 아니라 foward left right back 4방면으로 이동하면서 dfs를 확인한다.
        public static void DFS(int x, int y, int n)
        {
            //방문을 했으면 해당 cnt가 표시가 되고
            visit[x, y] = cnt;
            visitCnt++;
           

            //왼쪽을 확인한다. x- 1의 값으로 이동할거임 visit이력이 없어야하고 nums이 1이여야한다.
            if (x - 1 >= 0)
            {
                if (visit[x-1,y] ==0 && nums[x-1,y] == 1)
                {
                    DFS(x-1, y, n);
                }
            }

            //아래를 확인한다.
            if (y +1 < n)
            {
                if (visit[x, y+1] == 0 && nums[x, y+1] == 1)
                {
                    DFS(x, y+1, n);
                }
            }
            //위를 확인한다.
            if (y - 1 >= 0)
            {
                if (visit[x, y - 1] == 0 && nums[x, y - 1] == 1)
                {
                    DFS(x, y - 1, n);
                }
            }
            //오른쪽을 확인한다.
            if (x + 1 < n)
            {
                if (visit[x + 1, y] == 0 && nums[x + 1, y] == 1)
                {
                    DFS(x + 1, y, n);
                }
            }
        }
    }

}
```
