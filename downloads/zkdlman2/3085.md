# 3085번 - 사탕 게임

**난이도: Silver 2**

## 문제
상근이는 어렸을 적에 "봄보니 (Bomboni)" 게임을 즐겨했다.
가장 처음에 N×N크기에 사탕을 채워 놓는다. 사탕의 색은 모두 같지 않을 수도 있다. 상근이는 사탕의 색이 다른 인접한 두 칸을 고른다. 그 다음 고른 칸에 들어있는 사탕을 서로 교환한다. 이제, 모두 같은 색으로 이루어져 있는 가장 긴 연속 부분(행 또는 열)을 고른 다음 그 사탕을 모두 먹는다.
사탕이 채워진 상태가 주어졌을 때, 상근이가 먹을 수 있는 사탕의 최대 개수를 구하는 프로그램을 작성하시오.

## 입력
첫째 줄에 보드의 크기 N이 주어진다. (3 ≤ N ≤ 50)
다음 N개 줄에는 보드에 채워져 있는 사탕의 색상이 주어진다. 빨간색은 C, 파란색은 P, 초록색은 Z, 노란색은 Y로 주어진다.
사탕의 색이 다른 인접한 두 칸이 존재하는 입력만 주어진다.

## 출력
첫째 줄에 상근이가 먹을 수 있는 사탕의 최대 개수를 출력한다.

## 예제 입력 1
```
3
CCP
CCP
PPC
```

## 예제 출력 1
```
3
```

## 예제 입력 2
```
4
PPPP
CYZY
CCPY
PPCC
```

## 예제 출력 2
```
4
```

## 예제 입력 3
```
5
YCPZY
CYZZP
CCPPP
YCYZC
CPPZZ
```

## 예제 출력 3
```
4
```

## 소스코드 (C#)
```cs

using System.Diagnostics;

namespace ConsoleApp3
{
    internal class Program
    {
        //난쟁이 키가 100
        //일곱난쟁이를 구해라
        //난쟁이 키가 주어지고 100을 넘지 않는 자연수, 아홉 명 모두 다르고 나오면 바로 출력

        public static int maxCount = 0;
        public static void Main(string[] args)
        {
            //사탕을 nxn
            //사탕색은 모두 같지 않을 수도 있다.
            //인접한 두칸을 골라서 교환한다.
            //가장 긴 연속 부분을 먹는다.
            int a = int.Parse(Console.ReadLine());
            char[,] nums = new char[51, 51];

            //값 저장하기
            for (int i = 0; i < a; i++)
            {
                string v = Console.ReadLine();

                for (int k = 0; k < a; k++)
                {
                    nums[i, k] = v[k];
                }
            }

            //가로 방향으로 변경
            for (int i = 0; i < a; i++)
            {
                for (int j = 0; j< a - 1; j++)
                {
                    //교환할거고 확인할거고 다시 원래대로 
                    swap(ref nums[i, j], ref nums[i, j + 1]);
                    check(a, nums);
                    swap(ref nums[i, j], ref nums[i, j+1]);
                }
            }

            //세로 방향으로 변경
            for (int i = 0; i < a; i++)
            {
                for (int j = 0; j < a - 1; j++)
                {
                    //교환할거고 확인할거고 다시 원래대로 
                    swap(ref nums[j, i], ref nums[j + 1, i]);
                    check(a, nums);
                    swap(ref nums[j, i], ref nums[j+1, i]);
                }
            }

            Console.WriteLine(maxCount);
        }

        private static void swap(ref char a, ref char b)
        {
            char c = a;
            a = b;
            b = c;          
        }

        private static void check(int maxLength, char[,] arr)
        {
            char c;
            //가로 행으로 바뀐 부분을확인함
            for (int i = 0; i < maxLength; i++)
            {
                c = arr[i, 0];
                int count = 1;
                for(int j=1; j < maxLength; j++)
                {
                    if (c == arr[i, j]) count++;
                    else
                    {
                        count = 1;
                    }
                    c = arr[i, j];
                    if (count > maxCount) maxCount = count;
                }
            }
            //가로 행으로 바뀐 부분을확인함
            for (int i = 0; i < maxLength; i++)
            {
                c = arr[0, i];
                int count = 1;
                for (int j = 1; j < maxLength; j++)
                {
                    if (c == arr[j, i]) count++;
                    else
                    {
                        count = 1;
                    }
                    c = arr[j, i];
                    if (count > maxCount) maxCount = count;
                }
            }                    
        }
    }

}
```
